import json

# =========================
# 初始状态
# =========================
def new_game():
    return {
        "current_scene": "gate",
        "world_state": {
            "ritual_progress": 0,
            "astre_awareness": 0
        },
        "player_state": {
            "sanity": 100
        },
        "inventory": [],
        "clues": [],
        "foreshadow_seen": [],
        "scene_states": {}
    }

state = new_game()

# =========================
# 工具函数
# =========================
def add_foreshadow(key):
    if key not in state["foreshadow_seen"]:
        state["foreshadow_seen"].append(key)
        print("（你隐约觉得这里有些不对劲……）")

def update_astre_awareness():
    state["world_state"]["astre_awareness"] = len(state["foreshadow_seen"]) // 2

# =========================
# 场景定义
# =========================
def scene_gate():
    print("\n【铁门】")
    print("你推开庄园铁门，雾气缓缓涌入。")

    add_foreshadow("gate_symbol")

    print("1. 进入小径")
    choice = input("> ")

    if choice == "1":
        state["current_scene"] = "path"


def scene_path():
    print("\n【小径】")
    print("鹅卵石路延伸向远处主楼，两侧枯树如同扭曲的人影。")

    add_foreshadow("tree_shadow")

    print("1. 前往主楼")
    print("2. 返回铁门")

    choice = input("> ")

    if choice == "1":
        state["current_scene"] = "hall"
    elif choice == "2":
        state["current_scene"] = "gate"


def scene_hall():
    print("\n【大厅】")
    print("墙上挂满画像……你隐约觉得它们长得一样。")

    add_foreshadow("portrait_same_face")

    print("1. 去管家房间")
    print("2. 去温室")

    choice = input("> ")

    if choice == "1":
        state["current_scene"] = "butler_room"
    elif choice == "2":
        state["current_scene"] = "greenhouse"


def scene_butler_room():
    print("\n【管家起居室】")
    print("房间陈旧，桌上账本摊开，抽屉露出照片。")

    if "photo_checked" not in state["scene_states"]:
        print("你看到一张照片：年轻的管家与一位男人站在一起……那张脸与你记忆中的主人一模一样。")
        add_foreshadow("photo_same_face")
        state["scene_states"]["photo_checked"] = True

    print("1. 返回大厅")
    choice = input("> ")

    if choice == "1":
        state["current_scene"] = "hall"


def scene_greenhouse():
    print("\n【温室工具房】")
    print("中央放着一个木箱，上面是三位密码锁。")

    if "box_opened" not in state["scene_states"]:
        print("墙上的工具排列似乎有规律……")
        add_foreshadow("tool_pattern")

    print("1. 输入密码")
    print("2. 返回大厅")

    choice = input("> ")

    if choice == "1":
        code = input("输入三位密码: ")

        if code == "188":
            print("箱子打开，里面是肥料，但摆放方式有些奇怪。")
            state["scene_states"]["box_opened"] = True
            state["world_state"]["ritual_progress"] += 1
        else:
            print("密码错误。")

    elif choice == "2":
        state["current_scene"] = "hall"


# =========================
# 结局
# =========================
def check_ending():
    if state["world_state"]["astre_awareness"] >= 3:
        print("\n【结局：被注视者】")
        print("你意识到他已经知道你知道了。")
        return True

    if len(state["foreshadow_seen"]) >= 4:
        print("\n【结局：真相】")
        print("你终于明白——这里的一切，从未改变。")
        return True

    return False


# =========================
# 主循环
# =========================
def game_loop():
    while True:
        update_astre_awareness()

        if check_ending():
            break

        scene = state["current_scene"]

        if scene == "gate":
            scene_gate()
        elif scene == "path":
            scene_path()
        elif scene == "hall":
            scene_hall()
        elif scene == "butler_room":
            scene_butler_room()
        elif scene == "greenhouse":
            scene_greenhouse()


# =========================
# 启动游戏
# =========================
if __name__ == "__main__":
    game_loop()
