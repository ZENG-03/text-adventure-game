import tkinter as tk
from PIL import Image, ImageTk

# =========================
# 游戏状态
# =========================
state = {
    "scene": "gate",
    "foreshadow": [],
    "ritual": 0
}

# =========================
# 主窗口
# =========================
root = tk.Tk()
root.title("庄园谜局")
root.geometry("900x600")

# 背景图
bg_label = tk.Label(root)
bg_label.place(x=0, y=0, relwidth=1, relheight=1)

# 文本
text = tk.Label(root, text="", wraplength=800, justify="left",
                bg="black", fg="white", font=("微软雅黑", 14))
text.pack(side="top", fill="x")

# 按钮区域
frame = tk.Frame(root, bg="black")
frame.pack(side="bottom", fill="x")

# =========================
# 工具函数
# =========================
def set_bg(name):
    try:
        img = Image.open(f"images/{name}.jpg")
        img = img.resize((900, 600))
        img = ImageTk.PhotoImage(img)
        bg_label.config(image=img)
        bg_label.image = img
    except:
        bg_label.config(image="")

def add_foreshadow(key):
    if key not in state["foreshadow"]:
        state["foreshadow"].append(key)
        text.config(text=text.cget("text") + "\n（你感觉哪里不对劲……）")

def clear_buttons():
    for widget in frame.winfo_children():
        widget.destroy()

def add_button(label, command):
    btn = tk.Button(frame, text=label, command=command)
    btn.pack(side="left", padx=10, pady=10)

# =========================
# 场景
# =========================
def scene_gate():
    set_bg("gate")
    text.config(text="你推开铁门，雾气涌入。")

    add_foreshadow("gate")

    clear_buttons()
    add_button("进入", lambda: change_scene("path"))

def scene_path():
    set_bg("path")
    text.config(text="小径延伸，树影像人。")

    add_foreshadow("tree")

    clear_buttons()
    add_button("前往大厅", lambda: change_scene("hall"))
    add_button("返回", lambda: change_scene("gate"))

def scene_hall():
    set_bg("hall")
    text.config(text="大厅中画像似乎是同一个人。")

    add_foreshadow("portrait")

    clear_buttons()
    add_button("管家房", lambda: change_scene("butler"))
    add_button("温室", lambda: change_scene("greenhouse"))
    add_button("地下", lambda: change_scene("basement"))

def scene_butler():
    set_bg("butler")
    text.config(text="照片里的人没有变老。")

    add_foreshadow("photo")

    clear_buttons()
    add_button("返回", lambda: change_scene("hall"))

def scene_greenhouse():
    set_bg("greenhouse")
    text.config(text="工具排列奇怪。")

    add_foreshadow("tools")

    clear_buttons()
    add_button("输入密码", open_password)
    add_button("返回", lambda: change_scene("hall"))

def open_password():
    win = tk.Toplevel(root)
    win.title("输入密码")

    entry = tk.Entry(win)
    entry.pack()

    def check():
        if entry.get() == "188":
            state["ritual"] += 1
            win.destroy()
            text.config(text="箱子打开了。")
        else:
            text.config(text="密码错误。")

    tk.Button(win, text="确认", command=check).pack()

def scene_basement():
    set_bg("basement")
    text.config(text="符文避开中间。")

    add_foreshadow("rune")

    clear_buttons()
    add_button("进入实验室", lambda: change_scene("lab"))
    add_button("返回", lambda: change_scene("hall"))

def scene_lab():
    set_bg("lab")
    text.config(text="实验记录重复。")

    add_foreshadow("lab")

    clear_buttons()
    add_button("前往仪式厅", lambda: change_scene("ritual"))

def scene_ritual():
    set_bg("ritual")
    text.config(text="输入异常数量：")

    clear_buttons()

    entry = tk.Entry(frame)
    entry.pack()

    def check():
        if entry.get() == str(len(state["foreshadow"])):
            state["ritual"] += 2
            change_scene("final")
        else:
            text.config(text="仪式失败")

    add_button("确认", check)

def scene_final():
    set_bg("final")

    f = len(state["foreshadow"])

    if f >= 6:
        result = "你看破了真相。"
    else:
        result = "你什么也没发现。"

    text.config(text="最终：" + result)

    clear_buttons()

# =========================
# 场景切换
# =========================
def change_scene(name):
    state["scene"] = name
    scenes[name]()

# =========================
# 场景表
# =========================
scenes = {
    "gate": scene_gate,
    "path": scene_path,
    "hall": scene_hall,
    "butler": scene_butler,
    "greenhouse": scene_greenhouse,
    "basement": scene_basement,
    "lab": scene_lab,
    "ritual": scene_ritual,
    "final": scene_final
}

# =========================
# 启动
# =========================
change_scene("gate")
root.mainloop()
