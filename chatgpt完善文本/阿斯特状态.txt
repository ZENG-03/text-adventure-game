🧠 2️⃣ 更新阿斯特阶段
def update_astre():
    f = len(state["foreshadow"])
    r = state["ritual"]

    if f >= 2:
        state["astre"]["stage"] = 1
    if f >= 4:
        state["astre"]["stage"] = 2
    if f >= 6 or r >= 2:
        state["astre"]["stage"] = 3

👁 4️⃣ 阿斯特出现函数
def astre_appear():
    update_astre()

    stage = state["astre"]["stage"]
    lines = ASTRE_DIALOGUE[stage]

    import random
    line = random.choice(lines)

    text.config(text="【阿斯特】\n" + line)

    clear_buttons()
    add_button("……", lambda: return_to_scene())
🔁 5️⃣ 返回原场景
