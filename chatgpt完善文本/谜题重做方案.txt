```js

      StateAPI.addItem("七音提示");
      return "【获得线索】：七音提示";
    }
  },
  desc: `琴盖内写着：“当七音归位，记忆将重现。”`,
  options: [{ text: "返回", target: "musicroom_entry" }]
};
```

---

## 🔵 解谜核心

```js
scenes["music_puzzle_main"] = {
  desc: `你需要恢复音符顺序。`,
  input: {
    placeholder: "输入音符顺序",
    validate: (val) => val === "E-D-C-A-G-B-F",

    onSuccess: () => {
      StateAPI.addItem("旋律徽章");
      StateAPI.setFlag("music_puzzle_solved");
    },

    failMsg: "旋律听起来不对……"
  },

  options: [
    { text: "查看线索", target: "music_clue_piano" },
    { text: "返回", target: "musicroom_entry" }
  ]
};
```

---

## 🟣 剧情触发（自动）

```js
if (StateAPI.getFlag("music_puzzle_solved") && !StateAPI.getFlag("music_story_triggered")) {
  triggerScene("music_story_scene");
}
```
