## 一、界面美化与动效

将动态背景抽离为独立组件
为了避免在 HomeView.vue 中混入大量 Canvas 动画逻辑（不同形式的语言），我们创建一个专门的 DynamicBackground.vue 组件，然后在 HomeView 中引用它。这样既保持了组件的单一职责，又便于维护和复用。

一、新建动态背景组件
创建 src/components/DynamicBackground.vue：

vue
<template>
  <div class="dynamic-bg">
    <canvas ref="canvasRef" class="bg-canvas"></canvas>
    <div class="candles">
      <div v-for="i in 7" :key="i" class="candle" :style="{ animationDelay: `${i * 0.5}s` }"></div>
    </div>
  </div>
</template>

<script setup>
import { ref, onMounted, onUnmounted } from 'vue';

const canvasRef = ref(null);
let animationId = null;
let particles = [];

onMounted(() => {
  initCanvas();
  window.addEventListener('resize', initCanvas);
});

onUnmounted(() => {
  if (animationId) cancelAnimationFrame(animationId);
  window.removeEventListener('resize', initCanvas);
});

const initCanvas = () => {
  const canvas = canvasRef.value;
  if (!canvas) return;
  const width = window.innerWidth;
  const height = window.innerHeight;
  canvas.width = width;
  canvas.height = height;

  const ctx = canvas.getContext('2d');
  
  // 初始化雾气粒子
  particles = [];
  for (let i = 0; i < 80; i++) {
    particles.push({
      x: Math.random() * width,
      y: Math.random() * height,
      radius: Math.random() * 40 + 15,
      opacity: Math.random() * 0.25,
      speedX: (Math.random() - 0.5) * 0.3,
      speedY: (Math.random() - 0.5) * 0.15
    });
  }

  function animate() {
    if (!canvas) return;
    ctx.clearRect(0, 0, width, height);
    
    // 绘制深色背景
    ctx.fillStyle = '#1a1a2e';
    ctx.fillRect(0, 0, width, height);
    
    // 绘制庄园剪影（简单梯形）
    ctx.fillStyle = '#2a2a3e';
    ctx.beginPath();
    ctx.moveTo(width * 0.2, height);
    ctx.lineTo(width * 0.3, height * 0.7);
    ctx.lineTo(width * 0.7, height * 0.7);
    ctx.lineTo(width * 0.8, height);
    ctx.fill();
    
    // 绘制窗户（七个小光点）
    for (let i = 0; i < 7; i++) {
      const x = width * (0.35 + i * 0.05);
      const y = height * 0.6;
      ctx.fillStyle = `rgba(212, 175, 55, ${0.3 + Math.sin(Date.now() * 0.002 + i) * 0.2})`;
      ctx.fillRect(x, y, 8, 12);
    }
    
    // 绘制雾气粒子
    particles.forEach(p => {
      ctx.beginPath();
      ctx.arc(p.x, p.y, p.radius, 0, Math.PI * 2);
      ctx.fillStyle = `rgba(220, 220, 230, ${p.opacity})`;
      ctx.fill();
      p.x += p.speedX;
      p.y += p.speedY;
      if (p.x < -100) p.x = width + 100;
      if (p.x > width + 100) p.x = -100;
      if (p.y < -100) p.y = height + 100;
      if (p.y > height + 100) p.y = -100;
    });
    
    animationId = requestAnimationFrame(animate);
  }
  
  animate();
};
</script>

三、优点
职责分离：HomeView 只负责布局和交互，背景动画完全由独立组件管理。

可复用：DynamicBackground 可在其他需要动态背景的页面（如游戏结束页）重复使用。

性能可控：Canvas 动画在组件卸载时会自动取消 requestAnimationFrame，避免内存泄漏。

易于调试：背景逻辑集中在一个文件中，修改时无需触及主视图。

```javascript
const typeText = async (htmlString) => {
  // 使用临时 div 解析标签，逐字渲染时保持标签闭合
  const tempDiv = document.createElement('div');
  tempDiv.innerHTML = htmlString;
  const nodes = [];
  // 递归提取文本节点和标签
  function extract(node) {
    if (node.nodeType === Node.TEXT_NODE) {
      nodes.push({ type: 'text', content: node.textContent });
    } else if (node.nodeType === Node.ELEMENT_NODE) {
      nodes.push({ type: 'tag', tagName: node.tagName, attrs: node.attributes });
      node.childNodes.forEach(extract);
      nodes.push({ type: 'tag', tagName: '/' + node.tagName });
    }
  }
  tempDiv.childNodes.forEach(extract);
```

### 2.3 存档预览

修改 `SaveLoadPanel.vue`，在存档信息中增加场景描述和游戏内时间：

```javascript
// 保存时额外存储摘要
const saveGame = (slot) => {
  const saveData = {
    profile: store.profile,
    run: store.run,
    timestamp: Date.now(),
    summary: {
      sceneName: getSceneName(store.run.current_scene_id),
      medalsCount: store.run.medals.length,
      gameTime: store.run.game_time // 如果有
    }
  };
  localStorage.setItem(`save_${slot}`, JSON.stringify(saveData));
};

// 读取时显示摘要
const getSaveInfo = (slot) => {
  const data = JSON.parse(localStorage.getItem(`save_${slot}`));
  if (!data) return null;
  return {
    slot,
    date: new Date(data.timestamp).toLocaleString(),
    sceneName: data.summary.sceneName,
    medals: data.summary.medalsCount,
    time: data.summary.gameTime ? `第${data.summary.gameTime.day}天 ${data.summary.gameTime.hour}:00` : ''
  };
};
```

```html
  <p v-if="getSaveInfo(slot).time">时间：{{ getSaveInfo(slot).time }}</p>
```

### 3.3 语音输入（Web Speech API）

在 `TextInputUI.vue` 中添加语音按钮：

```vue
<template>
  <div class="input-group">
    <input v-model="userInput" />
    <button @click="startVoiceInput" v-if="isMobile">🎤</button>
  </div>
</template>

<script setup>
const recognition = ref(null);

const startVoiceInput = () => {
  if (!('webkitSpeechRecognition' in window)) {
    alert('您的浏览器不支持语音输入');
    return;
  }
  recognition.value = new webkitSpeechRecognition();
  recognition.value.lang = 'zh-CN';
  recognition.value.onresult = (event) => {
    userInput.value = event.results[0][0].transcript;
  };
  recognition.value.start();
};
</script>
```
